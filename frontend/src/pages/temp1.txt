  // 濡傛灉鏄暟瀛楋紝杞崲涓轰腑鏂?

    return STATUS_MAP[Number(status)] || `鐘舵€?{status}`;
  }
  // 濡傛灉宸茬粡鏄腑鏂囷紝鐩存帴杩斿洖
  return status;
};

  const [detailDrawerVisible, setDetailDrawerVisible] = useState(false);
  const [selectedOrder, setSelectedOrder] = useState(null);
  const [editModalVisible, setEditModalVisible] = useState(false);

      Object.keys(params).forEach(key => {
        if (params[key] === '' || params[key] === undefined || params[key] === null) {

      const data = await response.json();
      
      if (data.success) {
        setOrders(data.data);
        setPagination(prev => ({
          ...prev,
          total: data.total
        }));
      }

      message.error('鑾峰彇澶囪揣璁㈠崟澶辫触: ' + error.message);

      const response = await fetch(`/api/stock-orders/${selectedOrder.id}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },

      if (data.success) {
        message.success('鏇存柊鎴愬姛');

            message.success('鍒犻櫎鎴愬姛');
            fetchOrders();
          } else {
            message.error(data.message);
          }
        } catch (error) {
          message.error('鍒犻櫎澶辫触: ' + error.message);
        }

  const handleQueryJitRelations = async (record) => {
    setJitLoading(true);
    try {
      const response = await fetch('/api/stock-orders/jit-relations', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          shopId: 1,
          orderNos: record.order_number,
          selectJitMother: 1
        })
      });

      const data = await response.json();
      if (data.success) {
        setJitRelations(data.data);
        setJitModalVisible(true);
      } else {
        message.error(data.message);
      }

      message.error('鏌ヨ澶辫触: ' + error.message);
    } finally {
      setJitLoading(false);
    }
  };

  const handleCreateOrder = async (values) => {
    setCreateOrderLoading(true);
    try {
      // 瑙ｆ瀽杈撳叆鐨凷KU鏁版嵁
      const lines = values.skuData.trim().split('\n');
      const paramList = [];
      
      for (const line of lines) {
        if (!line.trim()) continue;
        
        const parts = line.trim().split(/\s+/);
        if (parts.length < 3) {
          message.error(`鏍煎紡閿欒: ${line}`);
          setCreateOrderLoading(false);
          return;
        }
        
        const skc = parts[0];
        const skuCode = parts[1];
        const orderCount = parseInt(parts[2]);
        
        if (isNaN(orderCount) || orderCount <= 0) {
          message.error(`鏁伴噺閿欒: ${line}`);
          setCreateOrderLoading(false);
          return;
        }
        
        // 鏌ユ壘鏄惁宸叉湁璇KC
        let skcItem = paramList.find(item => item.skc === skc);
        if (!skcItem) {
          skcItem = {
            skc: skc,
            skuCodeList: []
          };
          paramList.push(skcItem);
        }
        
        // 娣诲姞SKU
        skcItem.skuCodeList.push({
          skuCode: skuCode,
          orderCount: orderCount
        });
      }
      
      if (paramList.length === 0) {
        message.error('璇疯緭鍏ユ湁鏁堢殑SKU鏁版嵁');
        setCreateOrderLoading(false);
        return;
      }
      
      console.log('鍒涘缓澶囪揣鍗曞弬鏁?', paramList);
      
      const response = await fetch('/api/create-stock-order', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          shopId: 1, // 闇€瑕佷粠涓婁笅鏂囪幏鍙?
          paramList: paramList
        })
      });
      
      const data = await response.json();

        message.success('澶囪揣鍗曞垱寤烘垚鍔燂紒');
        setCreateOrderModalVisible(false);
        createOrderForm.resetFields();
        // 鍒锋柊鍒楄〃
        setTimeout(() => {
          fetchOrders();
        }, 2000);
      } else {
        message.error(data.message || '鍒涘缓澶辫触');
      }
    } catch (error) {
      message.error('鍒涘缓澶辫触: ' + error.message);
    } finally {
      setCreateOrderLoading(false);
    }

  const handlePrintMenuClick = async (type) => {
    if (selectedOrders.length === 0) {
      message.warning('璇峰厛閫夋嫨瑕佹墦鍗扮殑璁㈠崟');
      return;

      await prepareBarcodeData();

    } else if (type === 'pickList') {
      // 鎵撳嵃鎷ｈ揣鍗?
      await generatePickListPDF();
    } else if (type === 'washLabel') {
      message.info('鎵撳嵃娲楁按鏍囧姛鑳藉紑鍙戜腑...');

      message.info('PDA鎵撳嵃鏉＄爜鍔熻兘寮€鍙戜腑...');
    } else if (type === 'waybill') {
      message.info('鎵撳嵃闈㈠崟鍔熻兘寮€鍙戜腑...');

  const generatePickListPDF = async () => {
    setPrintLoading(true);
    message.loading('姝ｅ湪鐢熸垚鎷ｈ揣鍗昉DF...', 0);
    
    try {
      // 鏀堕泦鎵€鏈塖KU鏁版嵁锛屾寜璁㈠崟+SKC鍒嗙粍
      const groupedItems = new Map();
      
      selectedOrders.forEach(order => {
        if (order.items && order.items.length > 0) {
          order.items.forEach(item => {
            const skc = item.skc || '';
            const groupKey = `${order.order_number}_${skc}`;
            
            if (!groupedItems.has(groupKey)) {
              groupedItems.set(groupKey, {
                orderNumber: order.order_number,
                skc: skc,
                supplierCode: item.supplier_code || '',
                productImage: item.product_image || item.sku_image || '',
                addTime: order.add_time || order.allocate_time || '',
                requestDeliveryTime: order.request_delivery_time || '',
                skuList: []
              });
            }
            
            groupedItems.get(groupKey).skuList.push({
              skuAttribute: item.sku_attribute || '',
              skuCode: item.sku_code || '',
              supplierCode: item.supplier_code || '',
              orderQuantity: item.order_quantity || 0
            });
          });
        }
      });
      
      const pickItems = Array.from(groupedItems.values());
      
      if (pickItems.length === 0) {
        message.destroy();
        message.warning('娌℃湁鍙墦鍗扮殑鎷ｈ揣鏁版嵁');
        setPrintLoading(false);
        return;
      }
      
      // 褰撳墠鎵撳嵃鏃堕棿
      const printTime = new Date().toLocaleString('zh-CN', {
        year: 'numeric',
        month: '2-digit',
        day: '2-digit',
        hour: '2-digit',
        minute: '2-digit',
        second: '2-digit'
      }).replace(/\//g, '-');
      
      // 鏋勫缓琛ㄦ牸鏁版嵁
      const tableBody = [
        // 琛ㄥご
        [
          { text: '搴忓彿', style: 'tableHeader', alignment: 'center' },
          { text: '璁㈠崟淇℃伅', style: 'tableHeader', alignment: 'center' },
          { text: '鍟嗗SKU', style: 'tableHeader', alignment: 'center' },
          { text: '灞炴€ч泦', style: 'tableHeader', alignment: 'center' },
          { text: '涓嬪崟鏁伴噺', style: 'tableHeader', alignment: 'center' },
          { text: '鎷ｈ揣鏁伴噺', style: 'tableHeader', alignment: 'center', fillColor: '#ffff00' }
        ]
      ];
      
      let itemIndex = 1;
      for (const item of pickItems) {
        const addTimeStr = item.addTime ? formatTime(item.addTime).substring(0, 19) : '-';
        const reqTimeStr = item.requestDeliveryTime ? formatTime(item.requestDeliveryTime).substring(0, 19) : '';
        const skuCount = item.skuList.length;
        const totalQuantity = item.skuList.reduce((sum, sku) => sum + sku.orderQuantity, 0);
        
        // 璁㈠崟淇℃伅鏂囨湰
        const orderInfo = [
          `SKC锛?{item.skc}`,
          `SKC璐у彿锛?{item.supplierCode}`,
          `璁㈠崟鍙凤細${item.orderNumber}`,
          `涓嬪崟鏃堕棿锛?{addTimeStr}`
        ];
        if (reqTimeStr) {
          orderInfo.push(`瑕佹眰鍙栦欢锛?{reqTimeStr}`);
        }
        
        if (skuCount === 1) {
          // 鍗曚釜SKU
          tableBody.push([
            { text: itemIndex.toString(), alignment: 'center' },
            { text: orderInfo.join('\n'), fontSize: 9 },
            { text: item.skuList[0].supplierCode || '-', alignment: 'center', fontSize: 10 },
            { text: item.skuList[0].skuAttribute || '-', alignment: 'center', fontSize: 11 },
            { text: item.skuList[0].orderQuantity.toString(), alignment: 'center', fontSize: 11, bold: true },
            { text: '', fillColor: '#ffff00' }
          ]);
        } else {
          // 澶氫釜SKU - 绗竴琛?
          tableBody.push([
            { text: itemIndex.toString(), alignment: 'center', rowSpan: skuCount + 1 },
            { text: orderInfo.join('\n'), fontSize: 9, rowSpan: skuCount + 1 },
            { text: item.skuList[0].supplierCode || '-', alignment: 'center', fontSize: 10 },
            { text: item.skuList[0].skuAttribute || '-', alignment: 'center', fontSize: 11 },
            { text: item.skuList[0].orderQuantity.toString(), alignment: 'center', fontSize: 11, bold: true },
            { text: '', fillColor: '#ffff00' }
          ]);
          
          // 鍚庣画SKU琛?
          for (let i = 1; i < skuCount; i++) {
            const sku = item.skuList[i];
            tableBody.push([
              {}, // 搴忓彿宸插悎骞?
              {}, // 璁㈠崟淇℃伅宸插悎骞?
              { text: sku.supplierCode || '-', alignment: 'center', fontSize: 10 },
              { text: sku.skuAttribute || '-', alignment: 'center', fontSize: 11 },
              { text: sku.orderQuantity.toString(), alignment: 'center', fontSize: 11, bold: true },
              { text: '', fillColor: '#ffff00' }
            ]);
          }
          
          // 鍚堣琛?
          tableBody.push([
            {}, // 搴忓彿宸插悎骞?
            {}, // 璁㈠崟淇℃伅宸插悎骞?
            { text: '鍚堣', alignment: 'center', colSpan: 2, bold: true },
            {},
            { text: totalQuantity.toString(), alignment: 'center', fontSize: 11, bold: true },
            { text: '', fillColor: '#ffff00' }
          ]);
        }
        
        itemIndex++;
      }
      
      // 瀹氫箟PDF鏂囨。
      const docDefinition = {
        pageSize: 'A4',
        pageMargins: [20, 60, 20, 40],
        header: {
          columns: [
            { text: '鎷ｈ揣鍗?, style: 'header', alignment: 'left', margin: [20, 20, 0, 0] },
            { text: `鎵撳嵃鏃堕棿锛?{printTime}`, style: 'subheader', alignment: 'right', margin: [0, 25, 20, 0] }
          ]
        },
        content: [
          {
            table: {
              headerRows: 1,
              widths: [30, '*', 80, 60, 50, 50],
              body: tableBody
            },
            layout: {
              fillColor: function (rowIndex) {
                return rowIndex === 0 ? '#f5f5f5' : null;
              },
              hLineWidth: function () { return 1; },
              vLineWidth: function () { return 1; },
              hLineColor: function () { return '#000000'; },
              vLineColor: function () { return '#000000'; }
            }
          }
        ],
        styles: {
          header: {

          console.log('PDF鐢熸垚鎴愬姛锛屽ぇ灏?', pdfBlob.size);
          
          const fileName = `鎷ｈ揣鍗昣${new Date().toISOString().slice(0, 10)}_${Date.now()}.pdf`;
          
          // 灏咮lob杞负base64
          const reader = new FileReader();
          const base64Data = await new Promise((resolve) => {
            reader.onloadend = () => {
              const base64 = reader.result.split(',')[1];
              resolve(base64);
            };
            reader.readAsDataURL(pdfBlob);
          });
          
          // 灏濊瘯涓婁紶鍒癘SS鏈嶅姟
          let useOss = false;
          const ossUrl = process.env.REACT_APP_OSS_URL || `http://${window.location.hostname}:3001`;
          try {
            const healthCheck = await fetch(`${ossUrl}/health`, { 
              method: 'GET',
              signal: AbortSignal.timeout(2000)
            });
            useOss = healthCheck.ok;
          } catch (e) {
            console.warn('OSS鏈嶅姟涓嶅彲鐢紝浣跨敤鏈湴棰勮');
            useOss = false;
          }
          
          if (useOss) {
            const ossResponse = await fetch(`${ossUrl}/upload/buffer`, {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({
                buffer: base64Data,
                fileName: fileName,
                category: 'pdf',
                expiresIn: 7200
              })
            });
            
            if (!ossResponse.ok) {
              throw new Error(`OSS鏈嶅姟鍝嶅簲閿欒: ${ossResponse.status}`);
            }
            
            const ossResult = await ossResponse.json();
            
            if (ossResult.success) {
              window.open(ossResult.data.signedUrl, '_blank');
              message.destroy();
              message.success(`鎷ｈ揣鍗曞凡鐢熸垚锛岄摼鎺ユ湁鏁堟湡鑷?${ossResult.data.expiresAt.replace('T', ' ').slice(0, 19)}`);
            } else {
              throw new Error(ossResult.message || 'OSS涓婁紶澶辫触');
            }
          } else {
            const pdfUrl = URL.createObjectURL(pdfBlob);
            window.open(pdfUrl, '_blank');
            setTimeout(() => URL.revokeObjectURL(pdfUrl), 2 * 60 * 60 * 1000);
            message.destroy();
            message.success('鎷ｈ揣鍗曞凡鐢熸垚锛堟湰鍦伴瑙堟ā寮忥級');
          }
          
          setPrintLoading(false);
        } catch (error) {
          message.destroy();
          message.error('涓婁紶PDF澶辫触: ' + error.message);
          console.error('涓婁紶PDF澶辫触:', error);
          setPrintLoading(false);
        }
      });
      
    } catch (error) {
      message.destroy();
      message.error('鐢熸垚鎷ｈ揣鍗曞け璐? ' + error.message);
      console.error('鐢熸垚鎷ｈ揣鍗曞け璐?', error);
      setPrintLoading(false);
    }
  };
